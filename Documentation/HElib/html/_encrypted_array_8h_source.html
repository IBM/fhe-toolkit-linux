1186 // plaintextAutomorph: Compute b(X) = a(X^k) mod Phi_m(X).
1187 template <typename RX, typename RXModulus>
1188 void plaintextAutomorph(RX& bb,
1189  const RX& a,
1190  long k,
1191  long m,
1192  const RXModulus& PhimX)
1193 {
1194  // compute b(X) = a(X^k) mod (X^m-1)
1195  if (k == 1 || deg(a) <= 0) {
1196  bb = a;
1197  return;
1198  }
1199 
1200  RX b;
1201  b.SetLength(m);
1202  NTL::mulmod_precon_t precon = NTL::PrepMulModPrecon(k, m);
1203  for (long j = 0; j <= deg(a); j++)
1204  b[NTL::MulModPrecon(j, k, m, precon)] = a[j]; // b[j*k mod m] = a[j]
1205  b.normalize();
1206 
1207  rem(bb, b, PhimX); // reduce modulo the m'th cyclotomic
1208 }
