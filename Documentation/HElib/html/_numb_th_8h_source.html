497 template <typename T, bool maxFlag>
498 long argminmax(std::vector<T>& v)
499 {
500  if (v.size() < 1)
501  return -1; // error: this is an empty array
502  unsigned long idx = 0;
503  T target = v[0];
504  for (unsigned long i = 1; i < v.size(); i++)
505  if (maxFlag) {
506  if (v[i] > target) {
507  target = v[i];
508  idx = i;
509  }
510  } else {
511  if (v[i] < target) {
512  target = v[i];
513  idx = i;
514  }
515  }
516  return (long)idx;
517 }

627 // Example: rotate by 1 means [0 1 2 3] -> [3 0 1 2]
628 // rotate by -1 means [0 1 2 3] -> [1 2 3 0]
629 template <typename T>
630 void rotate(NTL::Vec<T>& v, long k)
631 {
632  long n = v.length();
633  if (n <= 1)
634  return;
635 
636  k %= n;
637  if (k < 0)
638  k += n;
639 
640  if (k == 0)
641  return;
642 
643  reverse(v, 0, n - 1);
644  reverse(v, 0, k - 1);
645  reverse(v, k, n - 1);
646 }
