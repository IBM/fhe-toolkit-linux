117  // NOTE: this is a bit heuristic: we assume that if we evaluate
118  // f at a primitive root of unity, then we get something that well
119  // approximates a normal random variable with the same variance,
120  // which is equal to the sum of the variances of the individual
121  // f_i's, which is (2*magBound)^2/12 = magBound^2/3.
122  // We then multiply the sqrt of the variance by scale to get
123  // the high probability bound.
124 
125  double noiseBoundForUniform(double magBound, long degBound) const
126  {
127  return scale * std::sqrt(double(degBound) / 3.0) * magBound;
128  }

133  }

144  // NOTE: for odd modulus, this means each f_i is uniformly distributed
145  // over { -floor(modulus/2), ..., floor(modulus/2) }.

147  // over { modulus/2, ..., modulus/2 }, except that the two endpoints
148  // (which represent the same residue class) occur with half the
149  // probability of the others.
150 
151  // NOTE: this is a bit heuristic: we assume that if we evaluate
152  // f at a primitive root of unity, then we get something that well
153  // approximates a normal random variable with the same variance,
154  // which is equal to the sum of the variances of the individual
155  // f_i's, which is (modulus)^2/12 + 1/6 for even modulus,
156  // and is at most (modulus^2)/12 for odd modulus.
157  // We then multiply the sqrt of the variance by scale to get
158  // the high probability bound.
159 
160  // NOTE: this is slightly more accurate that just calling
161  // noiseBoundForUniform with magBound=modulus/2.
162 
163  double noiseBoundForMod(long modulus, long degBound) const
164  {
165  double var = fsquare(modulus) / 12.0;
166  if (modulus % 2 == 0)
167  var += 1.0 / 6.0;
168 
169  return scale * std::sqrt(degBound * var);
170  }

182  // then we get a normal random variable variance degBound * sigma^2.
183  // We then multiply the sqrt of the variance by scale to get
184  // the high probability bound.

186  double noiseBoundForGaussian(double sigma, long degBound) const
187  {

250  double stdDevForRecryption(long skHwt = 0) const
251  {
252  if (!skHwt)
253  skHwt = rcData.skHwt;
254  // the default reverts to rcData.skHwt, *not* rcData.defSkHwt
255 
256  long k = zMStar.getNFactors();
257  // number of prime factors of m
258 
259  long m = zMStar.getM();
260  long phim = zMStar.getPhiM();
261 
262  double mrat = double(phim) / double(m);
263 
264  return std::sqrt(mrat * double(skHwt) * double(1L << k) / 3.0) * 0.5;
265  }
